# --- M0: Cloud ASR service (vocal wav -> LRC/ASS) ---
# Pinned loosely; correctness-before-speed. CPU path must always work.

# Pin the web framework so a clean CI resolve is reproducible. Unpinned, pip
# pulled starlette >=1.3, whose TestClient dropped the genuine `httpx` backend
# and demands a separate package — a supply-chain hazard in a credentialed CI.
# These are the versions the suite is developed/tested against; starlette 1.2.x
# uses real httpx (installed as a test-only dep in the deploy workflow).
fastapi==0.136.3
starlette==1.2.1
uvicorn[standard]==0.49.0
python-multipart           # multipart/form-data file upload

faster-whisper             # CPU/GPU ASR with int8 quantization (cheap server)
whisperx                   # word/char-level forced alignment (wav2vec2)
# whisperx pulls transformers (4.57.x), which hard-pins huggingface-hub<1.0. But
# pyannote (a transitive whisperx dep) otherwise drags in huggingface-hub 1.x,
# which makes `import transformers` raise -> whisperx align never loads and ALL
# word timing silently degrades to rough sentence-level interpolation
# (aligned=false). Pin <1.0 so forced alignment actually loads. Remove only when
# whisperx/transformers support huggingface-hub 1.x.
huggingface-hub>=0.34,<1.0
pythainlp                  # Thai word tokenization (newmm) -- REQUIRED: Thai has no spaces

pydantic
soundfile                  # read wav duration / sanity checks
numpy

# Phase A: verify Google Sign-In ID tokens (JWKS signature + aud/exp), with the
# certs cached by the library. Pure-python deps (rsa/pyasn1/cachetools); only the
# Modal `web` container actually calls it, but it's a light, app-wide dep.
google-auth
